const User = require("../Model/userModel"); // add at the top if missing

// ğŸš¨ Temporary route to create an admin (remove after first use!)
router.post("/seed-admin", async (req, res) => {
  try {
    const adminExists = await User.findOne({ role: "Admin" });
    if (adminExists) {
      return res.status(400).json({ message: "Admin already exists" });
    }

    const admin = new User({
      std_index: "A001",
      name: "Super Admin",
      bday: new Date("1990-01-01"),
      age: 35,
      address: "Colombo",
      password: "admin123", // plain text, will be hashed automatically
      role: "Admin",
      email: "jchanukamr@gmail.com" // <-- required for OTP
    });

    await admin.save();
    res.status(201).json({ message: "Admin created successfully", admin });
  } catch (err) {
    console.error(err);
    res.status(500).json({ message: "Error creating admin" });
  }
});
